Usage

app.js as your entry for example:

import { h5sdk, tt } from 'lark-client-api';

// in WebView
window.h5sdk.error(console.error);

h5sdk
    .config({
        // your configuration
    })
    .then(() => h5sdk.ready())
    .then(() => {
        // your logic
    });

    // your configuration
}).then(() => {
    // your logic
});
